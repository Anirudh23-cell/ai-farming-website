Same task, two ways

    
// count lines in all .log files

    

      

        

          

          count_logs.sh
        

        

          
#!/bin/bash
for f in *.log; do
  lines=$(wc -l < "$f")
  echo "$f: $lines lines"
done

        

      

      

        

          

          count_logs.py
        

        

          
import glob

for f in glob.glob("*.log"):
    with open(f) as fh:
        lines = len(fh.readlines())
    print(f"{f}: {lines} lines")
